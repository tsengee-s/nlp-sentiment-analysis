def get_dominant_topic(lda_model, dictionary, document):
    doc_bow = dictionary.doc2bow(document)
    topics = lda_model.get_document_topics(doc_bow)
    dominant_topic = max(topics, key=lambda x: x[1])
    return dominant_topic[0]
